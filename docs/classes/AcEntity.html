export class AcEntity {

	/**
	 * Creates entity from a json
	 * @param json
	 * @returns {AcEntity} entity
	 */
	static create(json?: any) {
		if (json) {
			return Object.assign(new AcEntity(), json);
		}
		return new AcEntity();
	}

	/**
	 * Creates entity from a json
	 * @param {Optional }json
	 * @returns {AcEntity} entity
	 */
	constructor(json?: any) {
		Object.assign(this, json);
	}
}
